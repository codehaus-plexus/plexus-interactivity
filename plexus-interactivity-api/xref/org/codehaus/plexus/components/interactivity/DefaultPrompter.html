27  import javax.inject.Inject;
28  import javax.inject.Named;
29  
30  import java.io.IOException;
31  import java.util.Iterator;
32  import java.util.List;
33  
34  import org.codehaus.plexus.util.StringUtils;
35  
36  /**
37   * Default prompter.
38   *
39   * @author Brett Porter
40   * @version $Id$
41   */
42  @Named
43  public class DefaultPrompter implements Prompter {
44      private final OutputHandler outputHandler;
45  
46      private final InputHandler inputHandler;
47  
48      @Inject
49      public DefaultPrompter(OutputHandler outputHandler, InputHandler inputHandler) {
50          this.outputHandler = outputHandler;
51          this.inputHandler = inputHandler;
52      }
53  
54      @Override
55      public String prompt(String message) throws PrompterException {
56          try {
57              writePrompt(message);
58          } catch (IOException e) {
59              throw new PrompterException("Failed to present prompt", e);
60          }
61  
62          try {
63              return inputHandler.readLine();
64          } catch (IOException e) {
65              throw new PrompterException("Failed to read user response", e);
66          }
67      }
68  
69      @Override
70      public String prompt(String message, String defaultReply) throws PrompterException {
71          try {
72              writePrompt(formatMessage(message, null, defaultReply));
73          } catch (IOException e) {
74              throw new PrompterException("Failed to present prompt", e);
75          }
76  
77          try {
78              String line = inputHandler.readLine();
79  
80              if (StringUtils.isEmpty(line)) {
81                  line = defaultReply;
82              }
83  
84              return line;
85          } catch (IOException e) {
86              throw new PrompterException("Failed to read user response", e);
87          }
88      }
89  
90      @Override
91      public String prompt(String message, List<String> possibleValues, String defaultReply) throws PrompterException {
92          String formattedMessage = formatMessage(message, possibleValues, defaultReply);
93  
94          String line;
95  
96          do {
97              try {
98                  writePrompt(formattedMessage);
99              } catch (IOException e) {
100                 throw new PrompterException("Failed to present prompt", e);
101             }
102 
103             try {
104                 line = inputHandler.readLine();
105                 if (line == null && defaultReply == null) {
106                     throw new IOException("EOF");
107                 }
108             } catch (IOException e) {
109                 throw new PrompterException("Failed to read user response", e);
110             }
111 
112             if (StringUtils.isEmpty(line)) {
113                 line = defaultReply;
114             }
115 
116             if (line != null && !possibleValues.contains(line)) {
117                 try {
118                     outputHandler.writeLine("Invalid selection.");
119                 } catch (IOException e) {
120                     throw new PrompterException("Failed to present feedback", e);
121                 }
122             }
123         } while (line == null || !possibleValues.contains(line));
124 
125         return line;
126     }
127 
128     @Override
129     public String prompt(String message, List<String> possibleValues) throws PrompterException {
130         return prompt(message, possibleValues, null);
131     }
132 
133     @Override
134     public String promptForPassword(String message) throws PrompterException {
135         try {
136             writePrompt(message);
137         } catch (IOException e) {
138             throw new PrompterException("Failed to present prompt", e);
139         }
140 
141         try {
142             return inputHandler.readPassword();
143         } catch (IOException e) {
144             throw new PrompterException("Failed to read user response", e);
145         }
146     }
147 
148     private String formatMessage(String message, List<String> possibleValues, String defaultReply) {
149         StringBuilder formatted = new StringBuilder(message.length() * 2);
150 
151         formatted.append(message);
152 
153         if (possibleValues != null && !possibleValues.isEmpty()) {
154             formatted.append(" (");
155 
156             for (Iterator<String> it = possibleValues.iterator(); it.hasNext(); ) {
157                 String possibleValue = it.next();
158 
159                 formatted.append(possibleValue);
160 
161                 if (it.hasNext()) {
162                     formatted.append('/');
163                 }
164             }
165 
166             formatted.append(')');
167         }
168 
169         if (defaultReply != null) {
170             formatted.append(' ').append(defaultReply);
171         }
172 
173         return formatted.toString();
174     }
175 
176     private void writePrompt(String message) throws IOException {
177         outputHandler.write(message + ": ");
178     }
179 
180     @Override
181     public void showMessage(String message) throws PrompterException {
182         try {
183             writePrompt(message);
184         } catch (IOException e) {
185             throw new PrompterException("Failed to present prompt", e);
186         }
187     }
188 }
